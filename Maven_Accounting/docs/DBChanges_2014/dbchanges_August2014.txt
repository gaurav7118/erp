ALTER TABLE `cheque`
ADD COLUMN `company`  varchar(255) NULL DEFAULT NULL;

ALTER TABLE `cheque` ADD FOREIGN KEY (`company`) REFERENCES `company` (`companyid`) ON DELETE RESTRICT ON UPDATE RESTRICT;

UPDATE cheque INNER JOIN paydetail ON cheque.id=paydetail.cheque SET cheque.company=paydetail.company;

ALTER TABLE `cheque`
ADD COLUMN `createdfrom` int NULL DEFAULT 1;

UPDATE cheque INNER JOIN paydetail pd ON pd.cheque = cheque.id INNER JOIN receipt rp ON rp.paydetail=pd.id SET createdfrom=2;


ALTER TABLE `cheque`
ADD COLUMN `sequencenumber`  int NULL DEFAULT 0;

ALTER TABLE `cheque`
ADD COLUMN `ischequenoautogenetated`  int(1) NULL DEFAULT 0;

ALTER TABLE `cheque`
ADD COLUMN `deleteflag`  int(1) NULL DEFAULT 0;

ALTER TABLE `cheque`
ADD COLUMN `bankaccount` varchar(255) NULL DEFAULT NULL;

ALTER TABLE `cheque` ADD FOREIGN KEY (`bankaccount`) REFERENCES `account` (`id`) ON DELETE RESTRICT ON UPDATE RESTRICT;

UPDATE cheque INNER JOIN paydetail pd ON pd.cheque = cheque.id INNER JOIN paymentmethod pm ON pm.id=pd.paymentMethod SET bankaccount=pm.account;
