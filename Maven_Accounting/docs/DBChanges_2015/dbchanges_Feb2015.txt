CREATE TABLE `budgeting` (
    `id` varchar(255) NOT NULL ,
    `department` varchar(255) DEFAULT NULL,
    `product` varchar(255) DEFAULT NULL,
    `productcategory` varchar(255) DEFAULT NULL,
    `frequencytype` varchar(255) DEFAULT NULL,
    `frequencycolumn` varchar(255) DEFAULT NULL,
    `amount` double DEFAULT '0',
    `year` varchar(255) DEFAULT NULL,
    `company`  varchar(255) DEFAULT NULL,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`department`) REFERENCES `masteritem` (`id`),
    FOREIGN KEY (`productcategory`) REFERENCES `masteritem` (`id`),
    FOREIGN KEY (`product`) REFERENCES `product` (`id`),
    FOREIGN KEY (`company`) REFERENCES `company` (`companyid`)
)
ENGINE=InnoDB
DEFAULT CHARACTER SET=utf8;
