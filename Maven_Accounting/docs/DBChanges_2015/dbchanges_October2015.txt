alter table invoice add amountduedate datetime;

#Invoice-Receipt

create view invoicereceiptview as 
select distinct inv.id as invoice ,max(je.entrydate) as jedate from receiptdetails rd  
inner join receipt on rd.receipt=receipt.id 
inner join journalentry je on je.id=receipt.journalentry 
inner join invoice inv on rd.invoice=inv.id 
where inv.id in (
select invoice.id from invoice where invoice.isdraft = 0 and 
((invoice.isopeningbalenceinvoice=0 and invoice.invoiceamountdue=0)
 or 
(invoice.isopeningbalenceinvoice=1 and invoice.openingbalanceamountdue=0))
) group by inv.id;


update invoice join invoicereceiptview on invoice.id = invoicereceiptview.invoice  set amountduedate=jedate;

drop view invoicereceiptview;

#Invoice-Link Receipt View

create view linkinvoicereceiptview as 
select distinct inv.id as invoice,FROM_UNIXTIME(max(receiptlinkdate)/1000) as jedate,lr.receipt from linkdetailreceipt lr  
inner join invoice inv on lr.invoice=inv.id
where  inv.id in (select invoice.id from invoice where invoice.isdraft = 0 and ((invoice.isopeningbalenceinvoice=0 and invoice.invoiceamountdue=0) or (invoice.isopeningbalenceinvoice=1 and invoice.openingbalanceamountdue=0))) group by inv.id;


update invoice join linkinvoicereceiptview on invoice.id = linkinvoicereceiptview.invoice  set amountduedate=jedate;

drop view linkinvoicereceiptview;

#Invoice-CN

create view invoicecnview as 
select distinct inv.id as invoice,max(je.entrydate) as jedate from cndetails cnd 
inner join creditnote on cnd.creditNote = creditnote.id  inner join journalentry  je on creditnote.journalentry=je.id  
inner join invoice inv on cnd.invoice=inv.id where inv.id in  (select invoice.id from invoice where invoice.isdraft = 0 and invoice.deleteflag=0 and ((invoice.isopeningbalenceinvoice=0 and invoice.invoiceamountdue=0 ) or ( invoice.isopeningbalenceinvoice=1 and invoice.openingbalanceamountdue=0 ))) group by inv.id;

update invoice join invoicecnview on invoice.id = invoicecnview.invoice  set amountduedate=jedate;

drop view invoicecnview;

#Invoice-Opening CN


create view openingcninvoice as 
select max(cn.creationdate) as jedate, 
inv.id invoice,cn.id creditnoteid from cndetails cnd 
 inner join creditnote cn on cnd.creditNote=cn.id  
inner join invoice inv on cnd.invoice=inv.id  where inv.amountduedate is null and ((inv.isopeningbalenceinvoice=0 
and inv.invoiceamountdue=0 ) or (inv.isopeningbalenceinvoice=1 and inv.openingbalanceamountdue=0)) and cn.creationdate is not null group by inv.id;

update invoice join openingcninvoice on invoice.id = openingcninvoice.invoice  set amountduedate=jedate;

drop view openingcninvoice;

# GoodsReceipt-Payment
# Store journalentry(entrydate) in goodsreceipt
create view grpaymentview as 
select distinct gr.id as goodsreceipt,max(je.entrydate) as entrydate from paymentdetail pdetail inner join payment on payment.id = pdetail.payment inner join journalentry je on je.id = payment.journalentry inner join goodsreceipt gr on gr.id=pdetail.goodsReceipt where gr.id in (select id from goodsreceipt where((goodsreceipt.isopeningbalenceinvoice=0 and goodsreceipt.invoiceamountdue=0) or (goodsreceipt.isopeningbalenceinvoice=1 and goodsreceipt.openingbalanceamountdue=0)) )  group by gr.id;

update goodsreceipt join grpaymentview on goodsreceipt.id = grpaymentview.goodsreceipt  set amountduedate=entrydate;

drop view grpaymentview;

# GoodsReceipt-Link Payment View


create view linkgrpaymentview as 
select  distinct goodsreceipt.id as goodsreceipt,FROM_UNIXTIME(max(lp.paymentlinkdate)/1000) as linkdate from linkdetailpayment lp inner join goodsreceipt on goodsreceipt.id=lp.goodsReceipt where goodsreceipt.id in (select id from goodsreceipt where((goodsreceipt.isopeningbalenceinvoice=0 and goodsreceipt.invoiceamountdue=0) or (goodsreceipt.isopeningbalenceinvoice=1 and goodsreceipt.openingbalanceamountdue=0)) and amountduedate is NULL) group by goodsreceipt.id;

update goodsreceipt join linkgrpaymentview on linkgrpaymentview.goodsreceipt=goodsreceipt.id set amountduedate=linkdate;

drop view linkgrpaymentview;

# GoodsReceipt-DN

create view grdebitnoteview as
select distinct gr.id as goodsreceipt,max(je.entrydate) as entrydate from dndetails dnd inner join debitnote dn on dnd.debitNote=dn.id inner join journalentry je on dn.journalentry=je.id inner join goodsreceipt gr on dnd.goodsreceipt=gr.id where gr.id in (select id from goodsreceipt where((goodsreceipt.isopeningbalenceinvoice=0 and goodsreceipt.invoiceamountdue=0) or (goodsreceipt.isopeningbalenceinvoice=1 and goodsreceipt.openingbalanceamountdue=0)) and amountduedate IS NULL )  group by gr.id; 


update goodsreceipt join grdebitnoteview on goodsreceipt.id = grdebitnoteview.goodsreceipt  set amountduedate=entrydate;

drop view grdebitnoteview;

# GoodsReceipt-Opening DN

create view openingdngrview as
select distinct gr.id as goodsreceipt,max(dn.creationdate) as creationdate from dndetails dnd inner join debitnote dn on dnd.debitNote=dn.id inner join goodsreceipt gr on dnd.goodsreceipt=gr.id where gr.id in (select id from goodsreceipt where((goodsreceipt.isopeningbalenceinvoice=0 and goodsreceipt.invoiceamountdue=0) or (goodsreceipt.isopeningbalenceinvoice=1 and goodsreceipt.openingbalanceamountdue=0)) and amountduedate IS NULL and dn.creationdate IS NOT NULL) group by gr.id; 

update goodsreceipt join openingdngrview on goodsreceipt.id = openingdngrview.goodsreceipt  set amountduedate=openingdngrview.creationdate;

drop view openingdngrview;
